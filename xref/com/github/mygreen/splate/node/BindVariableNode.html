16  package com.github.mygreen.splate.node;
17  
18  import org.springframework.core.style.ToStringCreator;
19  import org.springframework.expression.EvaluationContext;
20  import org.springframework.expression.Expression;
21  
22  import com.github.mygreen.splate.type.SqlTemplateValueType;
23  
24  import lombok.Getter;
25  
26  /**
27   * （コメントによる定義の）バインド変数のための{@link Node}です。
28   *
29   * @version 0.2
30   * @author higa
31   * @author T.TSUCHIE
32   */
33  public class BindVariableNode extends AbstractNode {
34  
35      /**
36       * 式
37       */
38      @Getter
39      private final String expression;
40  
41      /**
42       * パース済みの式
43       */
44      private final Expression parsedExpression;
45  
46      /**
47       * {@link BindVariableNode} を作成します。
48       *
49       * @param position テンプレートの位置情報
50       * @param expression 式
51       * @param parsedExpression パース済みの式
52       */
53      public BindVariableNode(final int position, final String expression, final Expression parsedExpression) {
54          super(position);
55          this.expression = expression;
56          this.parsedExpression = parsedExpression;
57      }
58  
59      @Override
60      public void accept(final NodeProcessContext ctx) {
61  
62          final EvaluationContext evaluationContext = ctx.getEvaluationContext();
63          Object value = evaluateExpression(parsedExpression, evaluationContext, Object.class, getPosition(), ctx.getParsedSql());
64          Class<?> clazz = parsedExpression.getValueType(evaluationContext);
65  
66          SqlTemplateValueType<?> valueType = ctx.getValueTypeRegistry().findValueType(clazz, expression);
67          value = getBindVariableValue(value, valueType, getPosition(), ctx.getParsedSql(), expression);
68          ctx.addSql("?", value);
69      }
70  
71      @Override
72      public String toString() {
73          return new ToStringCreator(this)
74                  .append("position", getPosition())
75                  .append("expression", expression)
76                  .append("parsedExpression", parsedExpression)
77                  .append("children", children)
78                  .toString();
79      }
80  }
