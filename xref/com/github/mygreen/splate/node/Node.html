16  package com.github.mygreen.splate.node;
17  
18  /**
19   * <code>SQL</code>を構成する個々の要素をあらわします。
20   *
21   * @author higa
22   *
23   */
24  public interface Node {
25  
26      /**
27       * 子供の要素数を返します。
28       *
29       * @return 要素数
30       */
31      int getChildSize();
32  
33      /**
34       * 指定されたインデックス番号<code>index</code>の<code>Node</code>を返します。
35       *
36       * @param index 子ノードのインデックス番号
37       * @return 子ノード
38       */
39      Node getChild(int index);
40  
41      /**
42       * 子供の<code>Node</code>を追加します。
43       *
44       * @param node 子ノード
45       */
46      void addChild(Node node);
47  
48      /**
49       * SQLテンプレートを実行し評価します。
50       *
51       * @param ctx SQLテンプレートを実行するときのコンテキスト。
52       */
53      void accept(NodeProcessContext ctx);
54  
55      /**
56       * テンプレート内での開始位置を返します。
57       *
58       * @return テンプレート内での開始位置
59       */
60      int getPosition();
61  }
