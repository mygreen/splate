36      @SuppressWarnings({"rawtypes", "unchecked"})
37      @Test
38      void testValueTypeRegistry() {
39  
40          // 元となる変換規則の定義
41          SqlTemplateValueTypeRegistry registry = new SqlTemplateValueTypeRegistry();
42          registry.register(Enum.class, new EnumOrdinalType());
43  
44  
45          String sql = "SELECT * FROM Employee emp WHERE job = /*job*/'CLERK'";
46  
47          SqlTemplate template = templateEngine.getTemplateByText(sql);
48  
49          MapSqlTemplateContext context = new MapSqlTemplateContext(registry, Map.of("job", JobType.COOKS));
50  
51          // 列挙型の変換規則を上書きする
52          context.registerValueType(Enum.class, new EnumNameType());
53  
54          ProcessResult result = template.process(context);
55  
56          assertThat(result.getSql()).isEqualTo("SELECT * FROM Employee emp WHERE job = ?");
57          assertThat(result.getParameters()).containsExactly("COOKS");
58  
59          // 元の変換規則のチェック - 変わっていないことをチェック
60          assertThat(registry.findValueType(JobType.class, null)).isInstanceOf(EnumOrdinalType.class);
61  
62      }
63  
64      @Test
65      void testEvaluationContextEditor() {
66  
67          String sql = "select * from Employee emp where /*IF #notEmpty(name)*/name like /*name*/'S%'/*END*/";
68  
69          SqlTemplate template = templateEngine.getTemplateByText(sql);
70          MapSqlTemplateContext templateContext = new MapSqlTemplateContext(Map.of("name", "%abc%"));
71  
72          // EL式中のカスタム関数の登録
73          templateContext.setEvaluationContextEditor(c -> {
74              try {
75                  c.registerFunction("notEmpty", SqlFunctions.class.getMethod("notEmpty", Object.class));
76              } catch (NoSuchMethodException | SecurityException e) {
77                  throw new RuntimeException(e);
78              }
79          });
80  
81          ProcessResult result = template.process(templateContext);
82  
83          assertThat(result.getSql()).isEqualTo("select * from Employee emp where name like ?");
84          assertThat(result.getParameters()).containsExactly("%abc%");
85  
86      }
87  
88      /**
89       * SQLテンプレート中で利用可能なカスタム関数
90       *
91       */
92      static class SqlFunctions {
93  
94          public static boolean notEmpty(Object value) {
95              return value != null && !value.toString().isEmpty();
96          }
97  
98      }
