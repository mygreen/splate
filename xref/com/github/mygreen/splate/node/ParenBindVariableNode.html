16  package com.github.mygreen.splate.node;
17  
18  import java.lang.reflect.Array;
19  import java.util.Collection;
20  import java.util.LinkedList;
21  
22  import org.springframework.core.style.ToStringCreator;
23  import org.springframework.expression.EvaluationContext;
24  import org.springframework.expression.Expression;
25  
26  import com.github.mygreen.splate.type.SqlTemplateValueType;
27  
28  import lombok.Getter;
29  
30  /**
31   * {@literal IN}のバインド変数用の{@link Node}です。
32   *
33   * @version 0.2
34   * @author higa
35   * @author shuji.w6e
36   * @author T.TSUCHIE
37   */
38  public class ParenBindVariableNode extends AbstractNode {
39  
40      /**
41       * 式
42       */
43      @Getter
44      private final String expression;
45  
46      /**
47       * パース済みの式
48       */
49      private final Expression parsedExpression;
50  
51      /**
52       * {@link ParenBindVariableNode} を作成します。
53       *
54       * @param position テンプレート位置情報
55       * @param expression 式
56       * @param parsedExpression パース済みの式
57       */
58      public ParenBindVariableNode(final int position, final String expression, final Expression parsedExpression) {
59          super(position);
60          this.expression = expression;
61          this.parsedExpression = parsedExpression;
62      }
63  
64      @SuppressWarnings("rawtypes")
65      @Override
66      public void accept(final NodeProcessContext ctx) {
67  
68          final EvaluationContext evaluationContext = ctx.getEvaluationContext();
69          final Object var = evaluateExpression(parsedExpression, evaluationContext, Object.class, getPosition(), ctx.getParsedSql());
70          if(var == null) {
71              return;
72          }
73  
74          if (var instanceof Collection) {
75              bindArray(ctx, Collection.class.cast(var).toArray());
76          } else if (var instanceof Iterable) {
77              bindArray(ctx, toArray(Iterable.class.cast(var)));
78          } else if (var.getClass().isArray()) {
79              bindArray(ctx, var);
80          } else {
81              // ただのオブジェクトの場合
82              Class<?> clazz = parsedExpression.getValueType(evaluationContext);
83              SqlTemplateValueType valueType = ctx.getValueTypeRegistry().findValueType(clazz, expression);
84              Object value = getBindVariableValue(var, valueType, getPosition(), ctx.getParsedSql(), expression);
85              ctx.addSql("?", value);
86          }
87  
88      }
89  
90      /**
91       * 配列に変換します。
92       * @param iterable
93       * @return
94       */
95      private Object[] toArray(Iterable<?> iterable) {
96          LinkedList<Object> list = new LinkedList<>();
97          for (Object o : iterable) {
98              list.add(o);
99          }
100         return list.toArray();
101     }
102 
103     /**
104      * 配列に変換してバインドする。
105      * @param ctx the NodeProcessContext
106      * @param array the variable array
107      */
108     @SuppressWarnings("rawtypes")
109     private void bindArray(final NodeProcessContext ctx, final Object array) {
110         int length = Array.getLength(array);
111         if (length == 0) {
112             return;
113         }
114         Class<?> clazz = null;
115         for (int i = 0; i < length; ++i) {
116             Object o = Array.get(array, i);
117             if (o != null) {
118                 clazz = o.getClass();
119             }
120         }
121 
122         ctx.addSql("(");
123         Object value = Array.get(array, 0);
124         SqlTemplateValueType valueType = ctx.getValueTypeRegistry().findValueType(clazz, expression);
125         value = getBindVariableValue(value, valueType, getPosition(), ctx.getParsedSql(), expression);
126 
127         ctx.addSql("?", value);
128         for (int i = 1; i < length; ++i) {
129             ctx.addSql(", ");
130             value = Array.get(array, i);
131             value = getBindVariableValue(value, valueType, getPosition(), ctx.getParsedSql(), expression);
132             ctx.addSql("?", value);
133         }
134         ctx.addSql(")");
135     }
136 
137     @Override
138     public String toString() {
139         return new ToStringCreator(this)
140                 .append("position", getPosition())
141                 .append("expression", expression)
142                 .append("parsedExpression", parsedExpression)
143                 .append("children", children)
144                 .toString();
145     }
146 }
