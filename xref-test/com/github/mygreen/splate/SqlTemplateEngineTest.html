18  class SqlTemplateEngineTest {
19  
20      private SqlTemplateEngine sqlTemplateEngine;
21  
22      @BeforeEach
23      void setUp() throws Exception {
24          this.sqlTemplateEngine = new SqlTemplateEngine();
25      }
26  
27      @Test
28      void testGetTemplateByLocation_cached() {
29  
30          sqlTemplateEngine.setCached(true);
31  
32          String path = "classpath:template/employee_select.sql";
33  
34          SqlTemplate template = sqlTemplateEngine.getTemplate(path);
35          SqlTemplate template2 = sqlTemplateEngine.getTemplate(path);
36  
37          // 結果をキャッシュしているので値は同じになる。
38          assertThat(template).isEqualTo(template2);
39  
40          // キャッシュをクリアした場合
41          sqlTemplateEngine.clearCache();
42          SqlTemplate template3 = sqlTemplateEngine.getTemplate(path);
43          assertThat(template).isNotEqualTo(template3);
44  
45      }
46
