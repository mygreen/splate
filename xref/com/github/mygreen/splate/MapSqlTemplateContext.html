1   package com.github.mygreen.splate;
2   
3   import java.util.HashMap;
4   import java.util.List;
5   import java.util.Map;
6   
7   import org.springframework.expression.spel.support.StandardEvaluationContext;
8   
9   import com.github.mygreen.splate.type.SqlTemplateValueTypeRegistry;
10  
11  import lombok.Getter;
12  import lombok.NonNull;
13  import lombok.Setter;
14  
15  /**
16   * SQLテンプレートのパラメータを{@link Map} として渡すときのSQLテンプレートのコンテキスト。
17   * SQLテンプレート中では、マップのキー名で参照できます。
18   *
19   *
20   * @version 0.3
21   * @author T.TSUCHIE
22   *
23   */
24  public class MapSqlTemplateContext extends SqlTemplateContext<StandardEvaluationContext> {
25  
26      private Map<String, Object> values = new HashMap<>();
27  
28      /**
29       * SQLテンプレート中に存在しないプロパティが定義されているとき、{@literal null} として無視するかどうか。
30       *
31       * @since 0.3
32       * @param ignoreNotFoundProperty SQLテンプレート中に存在しないプロパティが定義されているとき、{@literal null} として無視するかどうか設定します。
33       * @return SQLテンプレート中に存在しないプロパティが定義されているとき、{@literal null} として無視するかどうか返します。
34       */
35      @Setter
36      @Getter
37      private boolean ignoreNotFoundProperty;
38  
39      /**
40       * コンストラクタ。
41       */
42      public MapSqlTemplateContext() {
43          super();
44      }
45  
46      /**
47       * テンプレート中の変数を指定してインスタンスを作成します。
48       * @param variables SQLテンプレート中のパラメータとして渡すマップ。
49       */
50      public MapSqlTemplateContext(@NonNull Map<String, Object> variables) {
51          super();
52          this.values.putAll(variables);
53      }
54  
55      /**
56       * {@link SqlTemplateValueTypeRegistry}を指定してインスタンスを作成します。
57       * @param valueTypeRegistry SQLテンプレートのパラメータの変換処理を管理する処理。
58       */
59      public MapSqlTemplateContext(SqlTemplateValueTypeRegistry valueTypeRegistry) {
60          super(valueTypeRegistry);
61      }
62  
63      /**
64       * {@link SqlTemplateValueTypeRegistry}とテンプレート中の変数を指定してインスタンスを作成します。
65       * @param valueTypeRegistry SQLテンプレートのパラメータの変換処理を管理する処理。
66       * @param variables SQLテンプレート中で使用可能な変数
67       */
68      public MapSqlTemplateContext(SqlTemplateValueTypeRegistry valueTypeRegistry, @NonNull Map<String, Object> variables) {
69          super(valueTypeRegistry);
70          this.values.putAll(variables);
71      }
72  
73      /**
74       * {@inheritDoc}
75       * @return {@link StandardEvaluationContext} のインスタンスを返します。
76       */
77      @Override
78      public StandardEvaluationContext createEvaluationContext() {
79          StandardEvaluationContext evaluationContext = new StandardEvaluationContext();
80          evaluationContext.setPropertyAccessors(List.of(
81                  new CustomMapAccessor(ignoreNotFoundProperty)
82                  ,new CustomReflectivePropertyAccessor(ignoreNotFoundProperty)));
83          evaluationContext.setRootObject(values);
84          return evaluationContext;
85      }
86  
87      /**
88       * SQLテンプレート中で使用可能な変数を追加します。
89       *
90       * @param name 変数名
91       * @param value 値
92       * @return 自身のインスタンス
93       */
94      public MapSqlTemplateContext addVariable(@NonNull String name, Object value) {
95          this.values.put(name, value);
96          return this;
97      }
98  
99      /**
100      * SQLテンプレート中で使用可能な変数を追加します。
101      *
102      * @param variables 変数のマップ
103      * @return 自身のインスタンス
104      */
105     public MapSqlTemplateContext addVariables(@NonNull Map<String, Object> variables) {
106         this.values.putAll(variables);
107         return this;
108     }
109 
110 
111 }
