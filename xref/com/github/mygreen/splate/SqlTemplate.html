1   package com.github.mygreen.splate;
2   
3   import com.github.mygreen.splate.node.Node;
4   import com.github.mygreen.splate.node.NodeProcessContext;
5   
6   import lombok.Getter;
7   import lombok.RequiredArgsConstructor;
8   
9   /**
10   * パースしたSQLテンプレート情報を保持します。
11   *
12   *
13   * @version 0.2
14   * @author T.TSUCHIE
15   *
16   */
17  @RequiredArgsConstructor
18  public class SqlTemplate {
19  
20      /**
21       * パースされたSQL。
22       * <p>正規化によりトリムや最後のセミコロン({@literal ;})が削除されたものです。</p>
23       */
24      @Getter
25      private final String sql;
26  
27      /**
28       * SQLノード
29       */
30      @Getter
31      private final Node node;
32  
33      /**
34       * SQLテンプレートを評価します。
35       *
36       * @param templateContext SQLテンプレートに渡すコンテキスト。
37       * @return SQLテンプレートを評価した結果。
38       */
39      public ProcessResult process(final SqlTemplateContext<?> templateContext) {
40  
41          final NodeProcessContext processContext = new NodeProcessContext(templateContext);
42          processContext.setParsedSql(sql);
43  
44          // SQLテンプレートを評価します。
45          node.accept(processContext);
46  
47          return new ProcessResult(processContext.getProcessedSql(), processContext.getBindParams());
48      }
49  }
