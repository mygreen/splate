1   package com.github.mygreen.splate.type;
2   
3   import org.springframework.lang.Nullable;
4   
5   /**
6    * SQLテンプレート中のパラメータ（変数）の型とJDBCの型を橋渡しするためのインタフェースです。
7    *
8    *
9    * @author T.TSUCHIE
10   * @param <T> マッピング対象の型
11   */
12  public interface SqlTemplateValueType<T> {
13  
14      /**
15       * SQLのパラメータ変数として値を取得します。
16       * <p>JDBCが対応していないタイプの場合は、対応している値に変換します。</p>
17       *
18       * @param value 変換する値
19       * @return SQLのパラメータ変数。
20       * @throws SqlTypeConversionException SQLで扱う型（JDBCの型）の変換にした場合にスローされます。
21       */
22      Object getBindVariableValue(@Nullable T value) throws SqlTypeConversionException;
23  
24      /**
25       * SQLに直接埋め込む値として文字列に変換します。
26       *
27       * @param value 変換する値。非nullが渡されます。
28       * @return 文字列に変換した値
29       * @throws TextConversionException 値を文字列への変換に失敗したときにストローされます。
30       */
31      default String getEmbeddedValue(final @Nullable T value) throws TextConversionException {
32          if(value == null) {
33              return null;
34          }
35          return value.toString();
36      }
37  }
