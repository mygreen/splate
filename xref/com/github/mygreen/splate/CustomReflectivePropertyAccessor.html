1   package com.github.mygreen.splate;
2   
3   import org.springframework.expression.AccessException;
4   import org.springframework.expression.EvaluationContext;
5   import org.springframework.expression.TypedValue;
6   import org.springframework.expression.spel.support.ReflectivePropertyAccessor;
7   
8   import lombok.RequiredArgsConstructor;
9   
10  @RequiredArgsConstructor
11  public class CustomReflectivePropertyAccessor extends ReflectivePropertyAccessor {
12  
13      /**
14       * 存在しないプロパティが指定された場合、無視するかどうか。
15       */
16      private final boolean ignoreNotFoundProperty;
17  
18      private static final String NOT_FOUNE_MESSAGE = "Neither getter method nor field found for property";
19  
20      @Override
21      public boolean canRead(EvaluationContext context, Object target, String name) throws AccessException {
22          if(ignoreNotFoundProperty) {
23              return true;
24          } else {
25              return super.canRead(context, target, name);
26          }
27      }
28  
29      @Override
30      public TypedValue read(EvaluationContext context, Object target, String name) throws AccessException {
31  
32          if(target == null && ignoreNotFoundProperty) {
33              return new TypedValue(null);
34          }
35  
36          try {
37              return super.read(context, target, name);
38          } catch(AccessException e) {
39              if(e.getMessage().contains(NOT_FOUNE_MESSAGE) && ignoreNotFoundProperty) {
40                  return new TypedValue(null);
41              } else {
42                  throw e;
43              }
44          }
45  
46      }
47  
48  
49  }
