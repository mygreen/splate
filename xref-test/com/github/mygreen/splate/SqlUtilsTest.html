19      @Test
20      void testIndexOfAny() {
21  
22          assertThat(SqlUtils.indexOfAny(null, 0, null, "a")).isEqualTo(-1);
23          assertThat(SqlUtils.indexOfAny("abc", 0, new AtomicReference<CharSequence>(), new String[0])).isEqualTo(-1);
24  
25          {
26              AtomicReference<CharSequence> foundStr = new AtomicReference<>();
27              int index = SqlUtils.indexOfAny("zzabyycdxx", 0, foundStr, "ab", "cd");
28              assertThat(index).isEqualTo(2);
29              assertThat(foundStr.get()).isEqualTo("ab");
30          }
31  
32          {
33              AtomicReference<CharSequence> foundStr = new AtomicReference<>();
34              int index = SqlUtils.indexOfAny("zzabyycdxx", 0, foundStr, "cd", "ab");
35              assertThat(index).isEqualTo(2);
36              assertThat(foundStr.get()).isEqualTo("ab");
37          }
38  
39          {
40              AtomicReference<CharSequence> foundStr = new AtomicReference<>();
41              int index = SqlUtils.indexOfAny("zzabyycdxx", 0, foundStr, "zab", "aby");
42              assertThat(index).isEqualTo(1);
43              assertThat(foundStr.get()).isEqualTo("zab");
44          }
45  
46          {
47              AtomicReference<CharSequence> foundStr = new AtomicReference<>();
48              int index = SqlUtils.indexOfAny("zzabyycdxx", 0, foundStr, "", "abc");
49              assertThat(index).isZero();
50              assertThat(foundStr.get()).isEmpty();
51          }
52  
53      }

55      @Test
56      void testResolveSqlPosition() {
57  
58          String sql = "SELECT * "
59                  + "\r\n" + " FROM"
60                  + "\r\n" + " EMPLOYEE"
61                  + "\r\n" + " WHERE"
62                  + "\r\n" + "  /*IF id != null*/"
63                  + "\r\n" + "  id = /*id*/10"
64                  + "\r\n" + "  /*END*/"
65                  ;
66  
67          {
68              // 先頭行
69              Position result = SqlUtils.resolveSqlPosition(sql, 7);
70              assertThat(result.getRow()).isEqualTo(1);
71              assertThat(result.getCol()).isEqualTo(7);
72              assertThat(result.getLine()).isEqualTo("SELECT * ");
73          }
74  
75          {
76              // 行の途中
77              Position result = SqlUtils.resolveSqlPosition(sql, 46);
78              assertThat(result.getRow()).isEqualTo(5);
79              assertThat(result.getCol()).isEqualTo(8);
80              assertThat(result.getLine()).isEqualTo("  /*IF id != null*/");
81          }

83      }
