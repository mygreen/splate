1   package com.github.mygreen.splate;
2   
3   import java.io.IOException;
4   import java.util.Optional;
5   
6   import org.springframework.core.io.Resource;
7   import org.springframework.core.io.ResourceLoader;
8   import org.springframework.util.StringUtils;
9   
10  /**
11   * SQLテンプレートのファイルを読み込む処理です。
12   * <p>{@literal suffixName} が指定されている場合、接尾語がついているリソースを優先して読み込みます。</p>
13   *
14   *
15   * @author T.TSUCHIE
16   *
17   */
18  public class TemplateLoader {
19  
20      /**
21       * リソースを指定して読み込む。
22       * @param resource リソース
23       * @param encoding 文字コード
24       * @return SQLファイルの内容
25       * @throws TwoWaySqlException テンプレートの読み込みに失敗した場合にスローされます。
26       */
27      public String loadByResource(final Resource resource, final String encoding) {
28          try {
29              return SqlUtils.readStream(resource.getInputStream(), encoding);
30          } catch(IOException e) {
31              throw new TwoWaySqlException("Fail load resource", e);
32          }
33      }
34  
35      /**
36       * リソースパスを指定して読み込む。
37       * {@literal suffixName} が指定されている場合、接尾語付きのリソースを優先して読み込みます。
38       *
39       * @param location リロースパス
40       * @param resourceLoader リロースローダー。
41       * @param encoding 文字コード
42       * @param suffixName リロースの接尾語
43       * @return SQLファイルの内容
44       * @throws TwoWaySqlException テンプレートの読み込みに失敗した場合にスローされます。
45       */
46      public String loadByLocation(final String location,
47              final ResourceLoader resourceLoader, final String encoding, final Optional<String> suffixName) {
48  
49          if(suffixName.isPresent()) {
50              // 接尾語付きパスの読み込み
51              String suffixedPath = convertPathWithSuffixed(location, suffixName.get());
52              try {
53                  Resource resource = resourceLoader.getResource(suffixedPath);
54                  if(resource.isReadable()) {
55                      return SqlUtils.readStream(resource.getInputStream(), encoding);
56                  }
57              } catch(IOException e) {
58                  // 読み込めない場合
59                  throw new TwoWaySqlException(String.format("Fail load file : %s", suffixedPath), e);
60  
61              }
62          }
63  
64          // 元々のパスでの読み込み
65          try {
66              Resource resource = resourceLoader.getResource(location);
67              if(resource.isReadable()) {
68                  return SqlUtils.readStream(resource.getInputStream(), encoding);
69              }
70  
71          } catch(IOException e) {
72              // 読み込めない場合
73              throw new TwoWaySqlException(String.format("Fail load file : %s", location), e);
74  
75          }
76  
77          // 読み込み対象のファイルが見つからない場合
78          throw new TwoWaySqlException(String.format("Not found or non-readable file : %s", location));
79  
80      }
81  
82      /**
83       * SQLのパスを接尾語付きのパスに変換する。
84       * @param location 変換対象のパス
85       * @param suffixName リロースの接尾語
86       * @return 変換したパス
87       */
88      protected String convertPathWithSuffixed(final String location, final String suffixName) {
89  
90          final StringBuilder sb = new StringBuilder(location.length());
91  
92          final String extension = StringUtils.getFilenameExtension(location);
93  
94          // 拡張子を除いたファイル名に方言名を付ける
95          sb.append(StringUtils.stripFilenameExtension(location))
96              .append("-").append(suffixName);
97  
98          // 拡張子がある場合は戻す。
99          if(!StringUtils.isEmpty(extension)) {
100             sb.append(".").append(extension);
101         }
102 
103         return sb.toString();
104 
105     }
106 }
